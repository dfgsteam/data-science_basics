#!/bin/bash

# Function to split chapters
split_chapter() {
    local chapter_num=$1
    local articles=("${!2}")
    local input_file="$chapter_num.txt"
    local output_dir="chapter_$chapter_num"

    mkdir -p "$output_dir"

    for ((i = 0; i < ${#articles[@]}; i++)); do
        local start_article="${articles[$i]}"
        local end_article="${articles[$(($i + 1))]}"
        local output_file="$output_dir/$start_article.txt"

        if [[ "$end_article" == "LAST_ARTICLE" ]]; then
            sed "/^Artikel $start_article\$/,\$!d" "$input_file" >"$output_file"
        else
            sed "/^Artikel $start_article\$/,/^Artikel $end_article\$/!d; /^Artikel $end_article\$/d" "$input_file" >"$output_file"
        fi
    done

    echo "Chapter $chapter_num done"
}
